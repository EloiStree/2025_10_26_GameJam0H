def received_byte_game_information(byte_array:bytes):
    # https://github.com/EloiStree/HelloLanCodeTournamentAbstraction/tree/main
    
    global byte_last_received_game_information
    byte_last_received_game_information = byte_array
    print("Received byte game information length:")
    print(len(byte_array))

    # if start and finish by 1 2 3   3 2 1 is a player position info
    if BulletHellGame.is_player_bytes_information(byte_array):
        print("Player position information detected.")
        # Process player position information here

    
    # if start and finish by 3 2 1   1 2 3 is a bullet information
    elif BulletHellGame.is_bullet_bytes_information(byte_array):
        print("Bullet information detected.")
        if BulletHellGame.is_bullet_bytes_instanciation_information(byte_array):
            print("Bullet instanciation information detected.")
            bullet_start:BulletCreationPoint = BulletHellGame.parse_bytes_to_bullet_start_point(byte_array)
            print(f"Bullet start point: {bullet_start}")
            print(f"Position: ({bullet_start.float3_start_position[0]}, {bullet_start.float3_start_position[1]}, {bullet_start.float3_start_position[2]})")
            print(f"Direction: ({bullet_start.float3_start_direction[0]}, {bullet_start.float3_start_direction[1]}, {bullet_start.float3_start_direction[2]})")
            print(f"Speed: {bullet_start.uint_speed_per_second_mm}  Size: {bullet_start.uint_size_mm}   ")

        if BulletHellGame.is_bullet_bytes_destruction_information(byte_array):
            print("Bullet destruction information detected.")
            bullet_destruction:BulletDestructionPoint = BulletHellGame.parse_bytes_to_bullet_destruction_point(byte_array)
            print(f"Bullet destruction point: {bullet_destruction}")
            print(f"Position: ({bullet_destruction.float3_end_position[0]}, {bullet_destruction.float3_end_position[1]}, {bullet_destruction.float3_end_position[2]})")
            print(f"Time since creation: {bullet_destruction.uint_time_since_creation_ms}")

    else:
        print("Unknown byte array format.")

def percent_minus1_to_1_to_value99(percent:float) -> int:
    if percent == 0.0:
        return 50
    # Convert from -1 to 1 range to 0 to 1 range
    normalized = (percent + 1.0) / 2.0
    # Convert to 1-99 range
    value99 = int(1 + normalized * 98.0)
    #print (f"Converted percent {percent} to value99 {value99}")
    return value99

def create_player_gamepad_axis_value(joystick_left_horizontal:float, joystick_left_vertical:float, joystick_right_horizontal:float, joystick_right_vertical:float) -> int:


    value = 1800000000 
    value += percent_minus1_to_1_to_value99(joystick_left_horizontal) * 1000000
    value += percent_minus1_to_1_to_value99(joystick_left_vertical) * 10000
    value += percent_minus1_to_1_to_value99(joystick_right_horizontal) * 100
    value += percent_minus1_to_1_to_value99(joystick_right_vertical)
    return int(value)
